- 订阅事件数据结构更改为链表

> 避免随着事件的增加，而数据长度增加

```ts
class Node<T> {
  data: T // 节点数据
  next: Node<T> | null // 指向下一个节点的指针
  constructor(data: T) {
    this.data = data
    this.next = null
  }
}

export class SinglyLinkedList<T> {
  private head: Node<T> | null // 链表的头节点
  private size: number // 链表的大小
  private current: Node<T> | null // 当前节点指针
  constructor(initialValues?: T[]) {}
}
```
